angular-input-highlight


<textarea ng-model="text" 
  highlight="format"
  highlight-onchange="listMarkers($markers)"></textarea>
  

$rootScope.format = {
  'rgba(255,0,0,0.5)' : /\d+/g,
  '#619FFF'           : /@[a-z0-9]+/g,
  '#25B520'           : '{{ format['#25B520'] }}'
};

$rootScope.listMarkers = function($markers) { 
  $rootScope.markers = $markers;
};
